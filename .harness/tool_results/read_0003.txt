92	// Derive CAPTURE_RAMP_TURNS from getCaptureRampMultiplier behavior
93	// The function returns 0 during the ramp period, then positive values after
94	function deriveCaptureRampTurns(): number {
95	  for (let turns = 0; turns < 50; turns++) {
96	    if (getCaptureRampMultiplier(turns) > 0) {
97	      return turns; // This equals CAPTURE_RAMP_TURNS + 1
98	    }
99	  }
100	  return 6; // fallback: CAPTURE_RAMP_TURNS + 1
101	}
102	const CAPTURE_RAMP_TURNS = deriveCaptureRampTurns() - 1;

152	  const turn = source.replay.turns[source.turnIndex] ?? source.replay.turns[0];
153	  const board = turn.snapshotEnd;
154	  const factions = buildReplayFactions(source.replay);
155	  const ownership = new Map<string, string>();
156	
157	  for (const city of board.cities) {
158	    ownership.set(`${city.q},${city.r}`, city.factionId);
159	  }
160	  for (const village of board.villages) {
161	    ownership.set(`${village.q},${village.r}`, village.factionId);
162	  }
163	
164	  const hexes = source.replay.map.hexes.map((hex) => ({
165	    key: hex.key,
166	    q: hex.q,
167	    r: hex.r,
168	    terrain: hex.terrain,
169	    visibility: 'visible' as const,
170	    ownerFactionId: ownership.get(hex.key) ?? null,
171	  }));
172	
173	  return {
174	    activeFactionId: board.factions[0]?.id ?? null,
175	    map: {
176	      width: source.replay.map.width,
177	      height: source.replay.map.height,
178	      hexes,
179	    },
180	    factions,
181	    units: board.units.map((unit) => {
182	      const chassisId = inferChassisId(unit.prototypeName);
183	      return {
184	        id: unit.id,
185	        factionId: unit.factionId,
186	        q: unit.q,
187	        r: unit.r,
188	        hp: unit.hp,
189	        maxHp: unit.maxHp,
190	        attack: unit.attack ?? 0,
191	        defense: unit.defense ?? 0,
192	        effectiveDefense: unit.defense ?? 0,
193	        range: unit.range ?? 1,
194	        movesRemaining: 0,
195	        movesMax: 0,
196	        acted: false,
197	        canAct: unit.id === source.selectedUnitId && source.reachableHexes.length > 0,
198	        isActiveFaction: unit.factionId === (board.factions[0]?.id ?? null),
199	        status: 'inactive' as const,
200	        prototypeId: unit.prototypeId,
201	        prototypeName: unit.prototypeName,
202	        chassisId,
203	        role: chassisId,
204	        spriteKey: getSpriteKeyForUnit(unit.factionId, unit.prototypeName, chassisId, undefined),
205	        facing: unit.facing ?? 0,
206	        visible: true,
207	      };
208	    }),
209	    cities: board.cities.map((city) => ({
210	      id: city.id,
211	      name: city.name,
212	      factionId: city.factionId,
213	      q: city.q,
214	      r: city.r,
215	      visible: true,
216	      remembered: true,
217	      besieged: city.besieged,
218	      wallHp: city.wallHp,
219	      maxWallHp: city.maxWallHp,
220	      turnsSinceCapture: city.turnsSinceCapture,
221	    })),
222	    villages: board.villages.map((village) => ({
223	      id: village.id,
224	      name: village.name,
225	      factionId: village.factionId,
226	      q: village.q,
227	      r: village.r,
228	      visible: true,
229	      remembered: true,
230	    })),
231	    improvements: [],
232	    overlays: {
233	      borders: buildBorderEdges(hexes, factions),
234	      reachableHexes: source.reachableHexes,
235	      attackHexes: source.attackHexes,
236	      pathPreview: source.pathPreview,
237	      queuedPath: source.queuedPath,
238	      lastMove: null,
239	    },
240	    visibility: {
241	      mode: 'full',
242	      activeFactionId: board.factions[0]?.id ?? null,
243	    },
244	  };
245	}

247	function buildPlayWorldViewModel(source: PlayWorldSource): WorldViewModel {
248	  const { state } = source;
249	  if (!state.map) {
250	    throw new Error('Cannot build play-mode world view without a map.');
251	  }
252	
253	  const factions = buildPlayFactions(state);
254	  const hexVisibility = buildHexVisibilityMap(state);
255	  const hexes = Array.from(state.map.tiles.values()).map((tile) => {
256	    const key = hexToKey(tile.position);
257	    return {
258	      key,
259	      q: tile.position.q,
260	      r: tile.position.r,
261	      terrain: tile.terrain,
262	      visibility: hexVisibility.get(key) ?? 'hidden' as const,
263	      ownerFactionId: getHexOwner(tile.position, state) ?? null,
264	    };
265	  });
266	
267	  const moveCounts = new Map<string, number>();
268	  const attackCounts = new Map<string, number>();
269	  for (const unit of state.units.values()) {
270	    moveCounts.set(
271	      unit.id,
272	      unit.factionId === state.activeFactionId && unit.status === 'ready'
273	        ? getPlayableMoves(state, unit, source.registry).length
274	        : 0,
275	    );
276	    attackCounts.set(
277	      unit.id,
278	      unit.factionId === state.activeFactionId && unit.status === 'ready'
279	        ? getAttackableEnemies(state, unit).length
280	        : 0,
281	    );
282	  }
283	
284	  return {
285	    activeFactionId: state.activeFactionId,
286	    map: {
287	      width: state.map.width,
288	      height: state.map.height,
289	      hexes,
290	    },
291	    factions,
292	    units: Array.from(state.units.values()).map((unit) => {
293	      const prototype = state.prototypes.get(unit.prototypeId as never);
294	      const chassisId = prototype?.chassisId ?? inferChassisId(prototype?.name ?? unit.prototypeId);
295	      const canAct = unit.factionId === state.activeFactionId
296	        && unit.status === 'ready'
297	        && unit.hp > 0
298	        && ((moveCounts.get(unit.id) ?? 0) > 0 || (attackCounts.get(unit.id) ?? 0) > 0);
299	      const faction = state.factions.get(unit.factionId);
300	      const factionDoctrine = faction
301	        ? resolveCapabilityDoctrine(state.research.get(unit.factionId as never), faction)
302	        : undefined;
303	      const unitTransport = getUnitTransport(unit.id, state.transportMap);
304	      const boardableTransportIds = unit.factionId === state.activeFactionId && unit.hp > 0
305	        ? Array.from(state.units.values())
306	          .filter((candidate) => candidate.factionId === unit.factionId && candidate.id !== unit.id)
307	          .filter((candidate) => canBoardTransport(state, unit.id, candidate.id, source.registry, state.transportMap))
308	          .map((candidate) => candidate.id)
309	        : [];
310	      const validDisembarkHexes = unitTransport
311	        ? getValidDisembarkHexes(state, unitTransport.transportId, source.registry, state.transportMap)
312	        : [];
313	      const canBrace = !!prototype
314	        && canAct
315	        && (canUseBrace(prototype) || factionDoctrine?.fortressTranscendenceEnabled === true)
316	        && hasAdjacentEnemy(state, unit);
317	      const canAmbush = !!prototype
318	        && canAct
319	        && canUseAmbush(prototype, getTerrainAt(state, unit.position))
320	        && !hasAdjacentEnemy(state, unit);
321	      const baseDefense = prototype?.derivedStats.defense ?? 0;
322	      // Compute effective defense including terrain, improvements, cities, villages
323	      const tile = state.map?.tiles.get(`${unit.position.q},${unit.position.r}`);
324	      const terrainDef = tile ? source.registry.getTerrain(tile.terrain) : undefined;
325	      const terrainMod = terrainDef?.defenseModifier ?? 0;
326	      let improvementBonus = 0;
327	      for (const [, improvement] of state.improvements) {
328	        if (improvement.position.q === unit.position.q && improvement.position.r === unit.position.r) {
329	          improvementBonus = improvement.defenseBonus ?? 0;
330	          break;
331	        }
332	      }
333	      if (improvementBonus === 0) {
334	        for (const [, city] of state.cities) {
335	          if (city.position.q === unit.position.q && city.position.r === unit.position.r) {
336	            improvementBonus = 1; // +100% in cities
337	            break;
338	          }
339	        }
340	      }
341	      if (improvementBonus === 0) {
342	        for (const [, village] of state.villages) {
343	          if (village.position.q === unit.position.q && village.position.r === unit.position.r) {
344	            improvementBonus = 0.5; // +50% in villages
345	            break;
346	          }
347	        }
348	      }
349	      const effectiveDefense = Math.max(1, Math.round(baseDefense * (1 + terrainMod + improvementBonus)));
350	      return {
351	        id: unit.id,
352	        factionId: unit.factionId,
353	        q: unit.position.q,
354	        r: unit.position.r,
355	        hp: unit.hp,
356	        maxHp: unit.maxHp,
357	        attack: prototype?.derivedStats.attack ?? 0,
358	        defense: baseDefense,
359	        effectiveDefense,
360	        range: prototype?.derivedStats.range ?? 1,
361	        movesRemaining: unit.movesRemaining,
362	        movesMax: unit.maxMoves,
363	        acted: unit.factionId === state.activeFactionId ? !canAct : false,
364	        canAct,
365	        isActiveFaction: unit.factionId === state.activeFactionId,
366	        status: unit.factionId === state.activeFactionId
367	          ? (unit.status === 'fortified' ? 'fortified' as const : canAct ? 'ready' as const : 'spent' as const)
368	          : 'inactive' as const,
369	        prototypeId: unit.prototypeId,
370	        prototypeName: prototype?.name ?? unit.prototypeId,
371	        chassisId,
372	        movementClass: thisChassisMovementClass(prototype?.chassisId, source.registry),
373	        role: prototype?.derivedStats.role,
374	        spriteKey: getSpriteKeyForUnit(unit.factionId, prototype?.name ?? unit.prototypeId, chassisId, prototype?.sourceRecipeId),
375	        facing: unit.facing ?? 0,
376	        visible: (hexVisibility.get(hexToKey(unit.position)) ?? 'hidden') !== 'hidden',
377	        veteranLevel: unit.veteranLevel,
378	        xp: unit.xp,
379	        nativeDomain: faction?.nativeDomain,
380	        learnedAbilities: unit.learnedAbilities?.map((a) => a.domainId),
381	        isStealthed: unit.isStealthed,
382	        poisoned: (unit.poisoned || (unit.poisonStacks ?? 0) > 0) || undefined,
383	        routed: unit.routed || undefined,
384	        preparedAbility: unit.preparedAbility,
385	        isSettler: prototype?.tags?.includes('settler') || undefined,
386	        canBrace: canBrace || undefined,
387	        canAmbush: canAmbush || undefined,
388	        isEmbarked: unitTransport !== undefined || undefined,
389	        transportId: unitTransport?.transportId ?? null,
390	        boardableTransportIds: boardableTransportIds.length > 0 ? boardableTransportIds : undefined,
391	        validDisembarkHexes: validDisembarkHexes.length > 0 ? validDisembarkHexes : undefined,
392	      };
393	    }),
394	    cities: Array.from(state.cities.values()).map((city) => ({
395	      id: city.id,
396	      name: city.name,
397	      factionId: city.factionId,
398	      q: city.position.q,
399	      r: city.position.r,
400	      visible: (hexVisibility.get(hexToKey(city.position)) ?? 'hidden') !== 'hidden',
401	      remembered: true,
402	      besieged: city.besieged,
403	      wallHp: city.wallHP,
404	      maxWallHp: city.maxWallHP,
405	      turnsSinceCapture: city.turnsSinceCapture,
406	    })),
407	    villages: Array.from(state.villages.values()).map((village) => ({
408	      id: village.id,
409	      name: village.name,
410	      factionId: village.factionId,
411	      q: village.position.q,
412	      r: village.position.r,
413	      visible: (hexVisibility.get(hexToKey(village.position)) ?? 'hidden') !== 'hidden',
414	      remembered: true,
415	    })),
416	    improvements: Array.from(state.improvements.values()).map((improvement) => ({
417	      id: improvement.id,
418	      type: improvement.type,
419	      q: improvement.position.q,
420	      r: improvement.position.r,
421	      ownerFactionId: improvement.ownerFactionId,
422	    })),
423	    overlays: {
424	      borders: buildBorderEdges(hexes, factions),
425	      reachableHexes: source.reachableHexes,
426	      attackHexes: source.attackHexes,
427	      pathPreview: source.pathPreview,
428	      queuedPath: source.queuedPath,
429	      lastMove: source.lastMove,
430	    },
431	    visibility: {
432	      mode: 'fogged',
433	      activeFactionId: state.activeFactionId,
434	    },
435	  };
436	}

522	    recentIntents: [],

528	      ? (() => {
529	          const economy = deriveResourceIncome(state, state.activeFactionId, registry);
530	          return {
531	            income: economy.supplyIncome,
532	            used: economy.supplyDemand,
533	            deficit: getSupplyDeficit(economy),
534	          };
535	        })()
536	      : null,
537	    exhaustion: state.activeFactionId
538	      ? (() => {
539	          const ex = state.warExhaustion.get(state.activeFactionId);
540	          return ex
541	            ? {
542	                points: ex.exhaustionPoints,
543	                productionPenalty: calculateProductionPenalty(ex.exhaustionPoints),
544	                moralePenalty: calculateMoralePenalty(ex.exhaustionPoints),
545	              }
546	            : { points: 0, productionPenalty: 0, moralePenalty: 0 };
547	        })()
548	      : null,
549	  };
550	}

552	function buildResearchChip(
553	  state: GameState,
554	  registry: RulesRegistry,
555	): { activeNodeName: string | null; progress: number | null; totalCompleted: number } | null {
556	  const factionId = state.activeFactionId;
557	  if (!factionId) return null;
558	  const research = state.research.get(factionId as never);
559	  const faction = state.factions.get(factionId as never);
560	  if (!research || !faction) return null;
561	
562	  // Find the active node across all domains
563	  let activeNodeName: string | null = null;
564	  let activeNodeCost = 0;
565	  const activeProgress = research.activeNodeId
566	    ? (research.progressByNodeId[research.activeNodeId as never] ?? 0)
567	    : null;
568	
569	  if (research.activeNodeId) {
570	    const domainId = research.activeNodeId.split('_t')[0];
571	    const domain = registry.getResearchDomain(domainId);
572	    if (domain) {
573	      activeNodeName = domain.nodes[research.activeNodeId]?.name ?? research.activeNodeId;
574	      activeNodeCost = domain.nodes[research.activeNodeId]?.xpCost ?? 0;
575	    }
576	  }
577	
578	  return {
579	    activeNodeName,
580	    progress: activeProgress !== null && activeNodeCost > 0 ? activeProgress / activeNodeCost : null,
581	    totalCompleted: research.completedNodes.length,
582	  };
583	}

585	function buildSettlementPreview(
586	  state: GameState,
587	  selected: ClientSelection,
588	  hoveredKey: string | null,
589	  world: WorldViewModel,
590	): SettlementPreviewViewModel | null {
591	  if (!state.map || selected?.type !== 'unit') {
592	    return null;
593	  }
594	
595	  const unit = state.units.get(selected.unitId as never);
596	  const prototype = unit ? state.prototypes.get(unit.prototypeId as never) : null;
597	  if (!unit || !prototype?.tags?.includes('settler')) {
598	    return null;
599	  }
600	
601	  const currentKey = hexToKey(unit.position);
602	  const reachableKeys = new Set(world.overlays.reachableHexes.map((entry) => entry.key));
603	  const previewKey = hoveredKey && (hoveredKey === currentKey || reachableKeys.has(hoveredKey))
604	    ? hoveredKey
605	    : currentKey;
606	  const previewTile = state.map.tiles.get(previewKey);
607	  if (!previewTile) {
608	    return null;
609	  }
610	
611	  const position = { q: previewTile.position.q, r: previewTile.position.r };
612	  const bonuses = evaluateCitySiteBonuses(state.map, position, 2);
613	  const requiresMove = previewKey !== currentKey;
614	  const blocker = getSettlementOccupancyBlocker(state, position);
615	  const unitCanFoundNow = unit.factionId === state.activeFactionId
616	    && unit.status === 'ready'
617	    && unit.movesRemaining === unit.maxMoves;
618	  const canFoundNow = !requiresMove && unitCanFoundNow && !blocker;
619	
620	  let blockedReason: string | undefined;
621	  if (requiresMove) {
622	    blockedReason = 'Move the settler here to found a city.';
623	  } else if (blocker) {
624	    blockedReason = formatSettlementOccupancyBlocker(blocker);
625	  } else if (unit.factionId !== state.activeFactionId) {
626	    blockedReason = 'Only the active faction can found a city.';
627	  } else if (unit.status !== 'ready' || unit.movesRemaining !== unit.maxMoves) {
628	    blockedReason = 'Settlers need full moves to found a city.';
629	  }
630	
631	  return {
632	    q: position.q,
633	    r: position.r,
634	    terrain: previewTile.terrain,
635	    canFoundNow,
636	    requiresMove,
637	    blockedReason,
638	    ...buildSettlementBonusSummary(bonuses),
639	  };
640	}

930	        return undefined; // No ramp in effect
931	      }
932	      // turnsUntilOutput: turns until production starts (> 0 means still at 0%)
933	      const turnsUntilOutput = rampMultiplier <= 0
934	        ? CAPTURE_RAMP_TURNS - city.turnsSinceCapture + 1
935	        : 0;
936	      // turnsUntilFull: turns until 100% output

939	        : 0;

941	        turnsSinceCapture: city.turnsSinceCapture,
942	        rampMultiplier,
943	        turnsUntilOutput: Math.max(0, turnsUntilOutput),
944	        turnsUntilFull: Math.max(0, turnsUntilFull),
945	      };
946	    })(),

1008	function buildBorderEdges(
1009	  hexes: Array<{ key: string; q: number; r: number; ownerFactionId: string | null; visibility: 'visible' | 'explored' | 'hidden' }>,
1010	  factions: FactionView[],
1011	): BorderEdgeView[] {
1012	  const factionColors = new Map(factions.map((faction) => [faction.id, faction.color]));
1013	  const hexMap = new Map(hexes.map((hex) => [hex.key, hex]));
1014	  const edges: BorderEdgeView[] = [];
1015	
1016	  for (const hex of hexes) {
1017	    if (!hex.ownerFactionId || hex.visibility === 'hidden') {
1018	      continue;
1019	    }
1020	
1021	    for (const direction of BORDER_DIRECTIONS) {
1022	      const neighbor = hexMap.get(`${hex.q + direction.dq},${hex.r + direction.dr}`);
1023	      if (neighbor?.ownerFactionId === hex.ownerFactionId) {
1024	        continue;
1025	      }
1026	
1027	      edges.push({
1028	        id: `${hex.key}:${direction.side}`,
1029	        q: hex.q,
1030	        r: hex.r,
1031	        side: direction.side,
1032	        factionId: hex.ownerFactionId,
1033	        color: factionColors.get(hex.ownerFactionId) ?? '#f7e7bf',
1034	      });
1035	    }
1036	  }
1037	
1038	  return edges;
1039	}

1178	export function buildResearchInspectorViewModel(
1179	  state: GameState,
1180	  registry: RulesRegistry,
1181	): ResearchInspectorViewModel | null {
1182	  const factionId = state.activeFactionId;
1183	  if (!factionId) return null;
1184	  const faction = state.factions.get(factionId as never);
1185	  if (!faction) return null;
1186	
1187	  const research = state.research.get(factionId as never);
1188	  if (!research) return null;
1189	
1190	  const nativeDomain = faction.nativeDomain ?? '';
1191	  const learnedDomains = faction.learnedDomains ?? [nativeDomain];
1192	  const allDomains = registry.getAllResearchDomains();
1193	  const progression = getDomainProgression(faction, research);
1194	
1195	  // Build node VMs across all research domains
1196	  const nodes: ResearchNodeViewModel[] = [];
1197	  for (const domainDef of allDomains) {
1198	    const domainId = domainDef.id;
1199	    const isNative = domainId === nativeDomain;
1200	    const isUnlocked = learnedDomains.includes(domainId);
1201	
1202	    for (const nodeDef of Object.values(domainDef.nodes)) {
1203	      const isCompleted = research.completedNodes.includes(nodeDef.id as never);
1204	      const isActive = research.activeNodeId === nodeDef.id;
1205	      const progress = research.progressByNodeId[nodeDef.id as never] ?? 0;
1206	
1207	      const prereqsMet = (nodeDef.prerequisites ?? []).every((prereqId) =>
1208	        research.completedNodes.includes(prereqId as never),
1209	      );
1210	
1211	      let nodeState: ResearchNodeViewState;
1212	      if (!isUnlocked) nodeState = 'locked';
1213	      else if (isCompleted) nodeState = 'completed';
1214	      else if (isActive) nodeState = 'active';
1215	      else if (!prereqsMet) nodeState = 'locked';
1216	      else nodeState = 'available';
1217	
1218	      const estimatedTurns =
1219	        nodeState === 'active' && research.researchPerTurn > 0
1220	          ? Math.ceil(Math.max(0, nodeDef.xpCost - progress) / research.researchPerTurn)
1221	          : null;
1222	
1223	      nodes.push({
1224	        nodeId: nodeDef.id,
1225	        name: nodeDef.name,
1226	        tier: nodeDef.tier ?? 1,
1227	        xpCost: nodeDef.xpCost,
1228	        discountedXpCost: null,
1229	        currentProgress: progress,
1230	        state: nodeState,
1231	        prerequisites: nodeDef.prerequisites ?? [],
1232	        prerequisiteNames: [],
1233	        unlocks: [],
1234	        qualitativeEffect: nodeDef.qualitativeEffect?.description ?? null,
1235	        estimatedTurns,
1236	        domain: domainId,
1237	        isNative,
1238	        isLocked: !isUnlocked,
1239	      });
1240	    }
1241	  }
1242	
1243	  // Build domain pips for all 10 research domains
1244	  const capabilitiesVms = allDomains.map((domainDef) => {
1245	    const domainId = domainDef.id;
1246	    const tier = getDomainTier(faction, domainId, research.completedNodes);
1247	    return {
1248	      domainId,
1249	      domainName: domainDef.name,
1250	      description: domainDef.name,
1251	      level: tier,
1252	      hasResearchTrack: true,
1253	      codified: learnedDomains.includes(domainId),
1254	      t1Ready: tier >= 1,
1255	      t2Ready: tier >= 2,
1256	    };
1257	  });
1258	
1259	  // Find active node info across domains
1260	  let activeNodeName: string | null = null;
1261	  let activeNodeCost: number | null = null;
1262	  let activeNodeProgress: number | null = null;
1263	
1264	  if (research.activeNodeId) {
1265	    const domainId = research.activeNodeId.split('_t')[0];
1266	    const domain = registry.getResearchDomain(domainId);
1267	    if (domain?.nodes[research.activeNodeId]) {
1268	      activeNodeName = domain.nodes[research.activeNodeId].name;
1269	      activeNodeCost = domain.nodes[research.activeNodeId].xpCost;
1270	      activeNodeProgress = research.progressByNodeId[research.activeNodeId as never] ?? 0;
1271	    }
1272	  }
1273	
1274	  // Simplified rate breakdown — flat base rate only
1275	  const totalRate = research.researchPerTurn;
1276	
1277	  return {
1278	    factionId,
1279	    activeNodeId: research.activeNodeId,
1280	    activeNodeName,
1281	    activeNodeProgress,
1282	    activeNodeXpCost: activeNodeCost,
1283	    completedCount: research.completedNodes.length,
1284	    totalNodes: nodes.length,
1285	    nodes,
1286	    capabilities: capabilitiesVms,
1287	    rateBreakdown: {
1288	      base: research.researchPerTurn,
1289	      detail: progression.canBuildLateTier
1290	        ? `${learnedDomains.length} domains unlocked · late-tier production available`
1291	        : progression.canBuildMidTier
1292	          ? `${learnedDomains.length} domains unlocked · mid-tier production available`
1293	          : `${learnedDomains.length} domain unlocked · base production only`,
1294	      total: totalRate,
1295	    },
1296	    hasKnowledgeDiscount: false,
1297	  };
1298	}
